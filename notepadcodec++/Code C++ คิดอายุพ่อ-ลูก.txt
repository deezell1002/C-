#include<iostream>
#include<string>
using namespace std;
int main()
{
	for (;;) //ให้วน loop เรื่อย  ๆ 
	{
	int a,b,c;
	cout << "enter father's ages : " ;
	cin >> a;								//รับค่าอายุพ่อ
	cout << "enter son's ages : ";
	cin >> b;								//รับค่าอายุลูก
	c = b*2;								//หาสองเท่าของอายุลูกไว้ก่อน เก็บไว้ในตัวแปร c
	if (a<b||a<0||b<0) cout << "to be die!!!" << endl;		//ถ้าอายุพ่อมากกว่าลูก หรืออายุพ่อน้อยกว่า 0 หรืออายุลูกน้อยกว่า 0 ไม่ให้รันโปรแกรมต่อ
	else if (c-a==0&&c==a) cout << "this year" << endl ;	//ถ้าสองเท่าของอายุลูก - อายุพ่อแล้วเท่ากับศูนย์ แสดงว่าปีนี้พ่อก็มีอายุสองเท่าลูกแล้ว
	else if (c-a>0&&c!=a) cout << c-a << "  years old father'll double the child" << endl; // ถ้าสองเท่าของอายุลูก - อายุพ่อแล้วมากกว่า 0 แสดงว่ายังเหลืออีกเท่านั้นพ่อถึงจะมีอายุสองเท่าของลูก
	else if (c-a<0&&c!=a) cout << (-(c-a)) << "  years pass" << endl; //ถ้าสองเท่าของอายุลูก - อายุพ่อแล้วน้อยกว่า 0 (คือติดลบ) แสดงว่าอายุพ่อมากกว่าสองเท่าของลูกแล้ว ก็ให้แสดงว่าอายุพ่อผ่านมาเท่าไหร่แล้วจากสองเท่าของลูก โดยค่าที่ได้จะติดลบก็ใส่ operator (-) ข้างหน้าเพื่อให้ค่ากลายเป็นบวก (เกรียนอ่ะ)
	}
	cout << endl;
	return(0);
}
