//โปรแกรมแปลงเลขฐานสองเป็นเลขฐานสิบ 4 หลัก //

#include <iostream>
using namespace std;
int main()
{
	int number,a1,a2,a3,a4;					//กำหนดตัวแปรให้กับเลขฐานสองที่จะรับมา และสร้างตัวแปรเพื่อรับค่าของแต่ละหลักที่คำนวณแล้วเพื่อนำมาบวกกัน//
	cout << "enter number : " ;
	cin >> number;							//รับค่าตัวเลขฐานสองมา 4 หลัก
	a1 = ((number-number%1000)/1000*8) ;	//คำนวณหาค่าตัวเลขหลักแรก แล้วนำมาคูณ 2 ยกกำลัง 3 (โดยการหารเอาให้เหลือตัวเลขหลักแรกตัวเดียวแล้วนำมาคูณ)
	a2 = ((number%1000)/100 * 4);			//คำนวณหาค่าตัวเลขหลักที่สอง แล้วนำมาคูณ 2 ยกกำลัง 2 (โดยการหารเอาให้เหลือตัวเลขหลักที่สองตัวเดียวแล้วนำมาคูณ)
	a3 = ((number%100)/10 * 2);				//คำนวณหาค่าตัวเลขหลักที่สาม แล้วนำมาคูณ 2 ยกกำลัง 1 (โดยการหารเอาให้เหลือตัวเลขหลักที่สามตัวเดียวแล้วนำมาคูณ)
	a4 = ((number%10)/1 * 1);				//คำนวณหาค่าตัวเลขหลักที่สี่ แล้วนำมาคูณ 2 ยกกำลัง 0 (โดยการหารเอาให้เหลือตัวเลขหลักสุดท้ายตัวเดียวแล้วนำมาคูณ)
	if (number>1111||number<0) cout << "It isn't Binary Code or more ,Please enter number again!!" << endl;
	if (number<=1111&&number>=0)cout << "Oct is :" << a1+a2+a3+a4 << endl;				//นำค่าของตัวเลขทั้งสามตัวมาบวกกันจะได้เลขฐาน 10)
	return(0);
}
