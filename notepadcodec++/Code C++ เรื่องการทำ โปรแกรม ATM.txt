#include <iostream>
using namespace std;
int main()
{
	int num ,b=3000  ;
	char choice;
	for (;;)
	{
		cout << endl << "Menu" <<endl;
		cout << "1) Deposit" << endl; 
		cout << "2) Withdraw" << endl;
		cout << "3) Check Balance" << endl;
		cout << "Enter choice : " ;
		cin >> choice ;

			switch (choice)
			{
			case '1' :
				cout << "Enter Amount to Deposit : ";
				cin >> num;
				b = b + num;
				cout << "Balance is : " << b << endl;
			break;
			case '2' :
				cout << "Enter Amount to Withdraw : " ;
				cin >> num;
				b = b - num;
				cout << "Balance is : " << b << endl;
			break;
			case '3' :
				cout << "Balance is: " << b<< endl;
			break;
			}
	}
	return(0);
}
